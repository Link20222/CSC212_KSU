                public void copyQ(LinkQueue<Integer> q1,LinkQueue<Integer> q2)
                {
                        int temp;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                q2.enqueue(temp);
                        }
                }

                public void reverse(LinkQueue<Integer> q1)
                {
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();
                        int temp;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                addFirst(q2,temp);
                        }

                        copyQ(q2,q1);
                }

                public void reverseWithStack(LinkQueue<Integer> q1)
                {
                        LinkStack<Integer> s1 = new LinkStack<Integer>();
                        int temp;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                s1.push(temp);
                        }

                        while(! s1.empty())
                        {
                                temp = s1.pop();
                                q1.enqueue(temp);
                        }
                }

                public void reversedCopy(LinkQueue<Integer> q1,LinkQueue<Integer> q2)
                {
                        int temp;
                        LinkQueue<Integer> q3 = new LinkQueue<Integer>();

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                q3.enqueue(temp);
                                addFirst(q2,temp);
                        }

                        copyQ(q3,q1);
                }
                
                public void copyLinkQueue(LinkQueue<Integer> q1 , LinkQueue<Integer> q2)
                {
                        LinkQueue<Integer> q3 = new LinkQueue<Integer>();
                        int temp;

                        copyQ(q1,q3);
                        
                        while(q3.length() != 0)
                        {
                                temp = q3.serve();
                                q1.enqueue(temp);
                                q2.enqueue(temp);                       
                        }
                }

                public int size(LinkQueue<Integer> q1)
                {
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();
                        int temp;
                        int s = 0;
                        
                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                s++;
                                q2.enqueue(temp);                       
                        }
                        
                        copyQ(q2,q1);
                        
                        return s;
                }       

                public boolean search(LinkQueue<Integer> q1,int target)
                {
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        int temp;
                        boolean flag = false;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                q2.enqueue(temp);

                                if (temp == target)
                                        flag = true;
                        }

                        copyQ(q2,q1);
                        return flag;
                }

                public int searchIndex(LinkQueue<Integer> q1,int target)
                {
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        int temp;
                        int i = 0;              
                        int loc = 0;
                        
                        while(q1.length() != 0)

                public boolean equalQueues(LinkQueue<Integer> q1,LinkQueue<Integer> q2)
                {
                        if (size(q1) != size(q2))
                                return false;
                        
                        LinkQueue<Integer> q3 = new LinkQueue<Integer>();
                        LinkQueue<Integer> q4 = new LinkQueue<Integer>();

                        int temp1,temp2;
                        boolean flag = true;

                        while(q1.length() != 0)
                        {
                                temp1 = q1.serve();
                                temp2 = q2.serve();

                                q3.enqueue(temp1);
                                q4.enqueue(temp2);

                                if (temp1 != temp2)
                                        flag = false;
                        }

                        copyQ(q3,q1);
                        copyQ(q4,q2);

                        return flag;
                }

                public boolean addIndex(LinkQueue<Integer> q1,int index,int newData)
                {
                        int s = size(q1);
                        
                        if (index < 1 && index > s)
                                return false;

                        if (index == s+1)
                                q1.enqueue(newData);
                        else
                        {
                                LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                                int i = 1;

                                while(q1.length() != 0)
                                {
                                        if (i == index)
                                                q2.enqueue(newData);

                                        q2.enqueue(q1.serve());
                                        
                                        i++;
                                }
                                
                                copyQ(q2,q1);
                        }
                        
                        return true;
                }
                
                public void addSorted(LinkQueue<Integer> q1,int newData)
                {
                        int temp;
                        
                        if (q1.length() == 0)
                                q1.enqueue(newData);
                        else
                        {
                                LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                                temp = q1.serve();

                                while(q1.length() != 0 && temp < newData)
                                {
                                        q2.enqueue(temp);
                                        temp = q1.serve();
                                }

                                if (q1.length() != 0 && temp < newData)
                                {
                                        q2.enqueue(temp);
                                        q2.enqueue(newData);
                                }
                                else
                                {
                                        q2.enqueue(newData);
                                        q2.enqueue(temp);
                                        copyQ(q1,q2);
                                }                       

                                copyQ(q2,q1);
                        }
                }

                public int deleteLast(LinkQueue<Integer> q1)
                {
                        if (q1.length() == 0)

                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();
                        
                        int temp = 0;
                        
                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                
                                if (q1.length() != 0)
                                        q2.enqueue(temp);
                        }
                        
                        copyQ(q2,q1);
                        return temp;
                }

                public boolean deleteTarget(LinkQueue<Integer> q1,int target)
                {
                        if (q1.length() == 0)
                                return false;

                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        int temp = 0;
                        boolean found = false;

                        while(q1.length() != 0 && ! found)
                        {
                                temp = q1.serve();

                                if (temp == target)
                                        found = true;
                                else
                                        q2.enqueue(temp);
                        }
                        
                        copyQ(q1,q2);
                        copyQ(q2,q1);

                        return found;
                }

                public int deleteByIndex(LinkQueue<Integer> q1,int index)
                {
                        if (q1.length() == 0 || index < 1 || index > size(q1))
                                return 0;

                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        int temp = 0;
                        
                        for (int i = 1 ; i < index ; i++)
                        {
                                temp = q1.serve();
                                q2.enqueue(temp);
                        }
                        
                        temp = q1.serve();
                        copyQ(q1,q2);
                        copyQ(q2,q1);

                        return temp;
                }

                public boolean deleteLastOccurence(LinkQueue<Integer> q1,int m)
                {
                        if (q1.length() == 0)
                                return false;

                        reverseWithStack(q1);
                        boolean found = deleteTarget(q1,m);
                        reverseWithStack(q1);

                        return found;
                }

                public int minimum(LinkQueue<Integer> q1)
                {
                        if (q1.length() == 0)
                                return 0;

                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        int min = q1.serve();
                        q2.enqueue(min);
                        int temp;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();
                                q2.enqueue(temp);

                                if (temp < min)
                                        min = temp;
                        }

                        copyQ(q2,q1);

                        return min;
                }

                public void sort(LinkQueue<Integer> q1)
                {
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();
                        int temp;
                        
                        while(q1.length() != 0)
                        {
                                temp = minimum(q1);
                                deleteTarget(q1,temp);
                                q2.enqueue(temp);
                        }
                        
                        copyQ(q2,q1);
                }

                public void swapFirstLast(LinkQueue<Integer> q1)
                {
                        if (q1.length() == 0 || size(q1) == 1)
                                return;

                        int first = 0,last = 0;
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        first = q1.serve();
                        int temp = 0;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();

                                if (q1.length() != 0)
                                        last = temp;
                                else
                                        q2.enqueue(temp);
                        }
                        
                        q1.enqueue(last);
                        copyQ(q2,q1);
                        q1.enqueue(first);
                }

                public void swapFirstLastDelete(LinkQueue<Integer> q1)
                {
                        if (q1.length() == 0 || size(q1) == 1)
                                return;

                        int first = 0,last = 0;
                        LinkQueue<Integer> q2 = new LinkQueue<Integer>();

                        first = q1.serve();
                        int temp = 0;

                        while(q1.length() != 0)
                        {
                                temp = q1.serve();

                                if (q1.length() != 0)
                                        last = temp;
                                else
                                        q2.enqueue(temp);
                        }

                        if (first == last)
                                copyQ(q2,q1);
                        else
                        {
                                q1.enqueue(last);
                                copyQ(q2,q1);
                                q1.enqueue(first);
                        }
                }
